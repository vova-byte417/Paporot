const $ = (s) => document.querySelector(s);
const $$ = (s) => document.querySelectorAll(s);

// ─── Tab switching ───────────────────────────────────────
function switchTab(tabName) {{
  $$('.tab-panel').forEach(p => p.classList.remove('active'));
  $$('.header-nav-btn').forEach(b => b.classList.remove('active'));
  const panel = document.getElementById('tab-' + tabName);
  if (panel) panel.classList.add('active');
  const btn = document.querySelector('[data-tab="' + tabName + '"]');
  if (btn) btn.classList.add('active');

  if (tabName === 'panorama') {{
    requestAnimationFrame(() => renderPanorama());
  }}
}}

$$('.header-nav-btn').forEach(btn => {{
  btn.addEventListener('click', () => switchTab(btn.dataset.tab));
}});

// ─── Tooltip helpers ─────────────────────────────────────
const tooltip = document.getElementById('d3Tooltip');
function showTooltip(html, evt) {{
  tooltip.innerHTML = html;
  tooltip.style.display = 'block';
  const x = evt.pageX + 16;
  const y = evt.pageY - 10;
  tooltip.style.left = Math.min(x, window.innerWidth - 380) + 'px';
  tooltip.style.top = y + 'px';
}}
function moveTooltip(evt) {{
  tooltip.style.left = Math.min(evt.pageX + 16, window.innerWidth - 380) + 'px';
  tooltip.style.top = (evt.pageY - 10) + 'px';
}}
function hideTooltip() {{ tooltip.style.display = 'none'; }}

// ═══════════════════════════════════════════════════════════
// SIDEBAR
// ═══════════════════════════════════════════════════════════
function renderSidebar() {{
  // Task history
  const taskContainer = document.getElementById('sidebarTasks');
  const eval = D.eval;
  const date = eval.date || eval.created_at.slice(0, 10);
  const hasAdded = D.symbols && D.symbols.added && D.symbols.added.length > 0;
  const hasRemoved = D.symbols && D.symbols.removed && D.symbols.removed.length > 0;
  let dotClass = 'mixed';
  if (hasAdded && !hasRemoved) dotClass = 'added';
  else if (!hasAdded && hasRemoved) dotClass = 'removed';

  const outcomeLabel = eval.outcome || 'unknown';
  const outcomeLower = outcomeLabel.toLowerCase();
  const outcomeColor = outcomeLower.includes('pass') ? 'var(--green)' :
    outcomeLower.includes('fail') ? 'var(--red)' : 'var(--amber)';

  taskContainer.innerHTML = `
    <div class="sidebar-task-item" style="border:1px solid var(--card-border);">
      <div class="task-dot ${{dotClass}}" style="box-shadow: 0 0 6px ${{outcomeColor}};"></div>
      <div class="task-info">
        <div class="task-date">${{date}}</div>
        <div class="task-desc">${{escapeHtml(eval.task.description || 'Untitled')}}</div>
        <div style="font-size:11px;color:var(--text-muted);margin-top:2px;">
          ${{D.filesChanged}} 文件 · ${{D.additions}}/+${{D.deletions}}/- · ${{outcomeLabel}}
        </div>
      </div>
    </div>
  `;

  if (D.prevTrials > 0) {{
    const info = document.createElement('div');
    info.style.cssText = 'font-size:11px;color:var(--text-muted);padding:8px 12px;';
    info.textContent = '+ ' + D.prevTrials + ' 条历史记录';
    taskContainer.appendChild(info);
  }}

  // Module index
  const modContainer = document.getElementById('sidebarModules');
  const allMods = (D.waterfall && D.waterfall.allModules) ? D.waterfall.allModules : (D.modules || []);
  if (allMods.length === 0) {{
    modContainer.innerHTML = '<span style="font-size:12px;color:var(--text-muted);">暂无模块数据</span>';
  }} else {{
    modContainer.innerHTML = '';
    const symsByMod = (D.waterfall && D.waterfall.symbolsByModule) ? D.waterfall.symbolsByModule : {{}};
    allMods.forEach(mod => {{
      const syms = symsByMod[mod] || [];
      const shortName = mod.split('/').pop() || mod;
      const tag = document.createElement('span');
      tag.className = 'sidebar-module-tag';
      tag.innerHTML = shortName + '<span class="tag-count">' + syms.length + '</span>';
      tag.title = mod + ' — ' + syms.length + ' 符号变更';
      tag.addEventListener('click', () => {{
        switchTab('panorama');
        setTimeout(() => focusPanoramaNode(mod), 500);
      }});
      modContainer.appendChild(tag);
    }});
  }}
}}
